	\begin{frame}{Model}
		 \begin{align*}
    	&H = H_J + H_D\\
   	 	& \text{Heisenberg exchange Hamiltonian}: H_J  = - J \sum_r \vec{S}_r \cdot \left(  \vec{S}_{r + \hat{x}} + \vec{S}_{r + \hat{y}}  \right) - H_Z \sum_r \vec{S}_r\\
    	& \text{Dzyaloshinskii-Moriya interaction Hamiltonian}: H_D = D \sum_r \vec{S}_r \times \vec{S}_{r + \hat{x}} \cdot \hat{x} +  \vec{S}_r \times \vec{S}_{r + \hat{y}} \cdot \hat{y}
 \end{align*}
	\end{frame}
